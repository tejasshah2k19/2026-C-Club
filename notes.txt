4) increment{++} and decrement{--} Operator 


int a = 10;
a++; // post increment  =>      a = a + 1;  
printf(" %d ",a); // 11 


int a = 10;
++a; //pre increment a = a + 1 
printf(" %d ",a); // 11 



int a = 10;
a--; // post decrement   a = a - 1 
printf(" %d ",a); // 9 


int a = 10;
--a; //pre DEcrement  a = a - 1 
printf(" %d ",a); // 9

int main() {
    int x = 5;

    int y = x++; //post increment y = 5  
    //x = 6   
    int z = ++x;//pre increment  x = 7 z = 7  

    printf("x = %d\n", x);//7 
    printf("y = %d\n", y);//5 
    printf("z = %d\n", z);//7 

    return 0;
}


#include<stdio.h>

int main() {
    int a = 20;

    int b = ++a;  // pre increment a = 21 b= 21 
    int c = a++;  // post increment c = 21 a = 22 

    printf("a = %d\n", a);//22 
    printf("b = %d\n", b);//21
    printf("c = %d\n", c);//21 

    return 0;
}
